class Solution {
    public List<Boolean> kidsWithCandies(int[] candies, int e) {
        List<Boolean> res = new ArrayList<>();
        int max=0;
        for(int i: candies){
            if(i>max) max=i;
        }
        for(int i:candies){
            if((i+e)>=max) res.add(true);
            else res.add(false);
        }
        return res;
        
    }
}
